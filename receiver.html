const express = require("express"); const path = require("path"); const app = express(); const server = require("http").createServer(app); const io = require("socket.io")(server); app.use(express.static(path.join(__dirname + "/public"))); const senderToReceiversMap = {}; // Store sender to receivers mapping io.on("connection", function (socket) { socket.on("sender-join", function (data) { socket.join(data.uid); senderToReceiversMap[data.uid] = []; // Initialize receivers array for sender }); socket.on("receiver-join", function (data) { socket.join(data.uid); socket.join(data.sender_uid); senderToReceiversMap[data.sender_uid].push(data.uid); // Add receiver to sender's array socket.in(data.sender_uid).emit("init", data.uid); }); socket.on("file-meta", function (data) { const senderID = Object.keys(socket.rooms).find(roomID => roomID !== socket.id); socket.in(data.uid).emit("fs-meta", { ...data, senderID }); // Send sender ID with metadata }); socket.on("fs-start", function (data) { socket.in(data.uid).emit("fs-share", { senderID: data.uid }); // Send sender ID for file sharing }); socket.on("file-raw", function (data) { socket.in(data.uid).emit("fs-share", data); }); }); server.listen(5000);
